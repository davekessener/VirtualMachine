DEA
	FROM A INITIAL   TO B WITH 0 TO E WITH 1
	FROM B           TO C WITH 0 TO F WITH 1
	FROM C ACCEPTING TO D WITH 0 TO H WITH 1
	FROM D           TO E WITH 0 TO H WITH 1
	FROM E           TO F WITH 0 TO I WITH 1
	FROM F ACCEPTING TO G WITH 0 TO B WITH 1
	FROM G           TO H WITH 0 TO B WITH 1
	FROM H           TO I WITH 0 TO C WITH 1
	FROM I ACCEPTING TO A WITH 0 TO E WITH 1
