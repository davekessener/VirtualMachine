PARSER TTable
VAR NAME = "[A-Z][A-Z0-9_]*"
VAR SETNAME = ""
VAR ADDVAR = ""
BEGIN
	RULE E  = O1 FINAL
	RULE O1 = ID(R$NAME) "=" HOOK($SETNAME) F1 O2
	RULE F1 = ID(R$NAME) "," HOOK($ADDVAR) F1
			= "="
	RULE O2 = O3 F2
	RULE F2 = "^" O3 PRINT("AND") F2
			= "v" O3 PRINT("OR") F2
			= EMPTY
	RULE O3 = "!" O4 PRINT("NOT")
			= O4
	RULE O4 = "(" O2 ")"
			= ID(R$NAME) PRINT
END
